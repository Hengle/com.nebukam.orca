71 protected override void Prepare(ref Unemployed job, float delta)
72 {
73
74 int agentCount = m_lockedAgents.Count;
75
76 if (m_outputAgents.Length != agentCount)
77 {
78 m_outputAgents.Dispose();
79 m_outputAgents = new NativeArray<AgentData>(agentCount, Allocator.Persistent);
80 }
81
82 m_maxRadius = 0f;
83
84 Agent a;
85 float3 pos, prefVel, vel;
86
87 if (plane == AxisPair.XY)
88 {
89 for (int i = 0; i < agentCount; i++)
90 {
91 a = m_lockedAgents[i];
92 m_maxRadius = max(m_maxRadius , a.radius);
93 pos = a.pos;
94 prefVel = a.m_prefVelocity;
95 vel = a.m_velocity;
96 m_outputAgents[i] = new AgentData()
97 {
98 index = i,
99 kdIndex = i,
100 position = float2(pos.x, pos.y), //
101 worldPosition = pos,
102 baseline = pos.z,
103 prefVelocity = float2(prefVel.x, prefVel.y),
104 velocity = float2(vel.x, vel.y),
105 worldVelocity = vel,
106 height = a.m_height,
107 radius = a.m_radius,
108 radiusObst = a.m_radiusObst,
109 maxSpeed = a.m_maxSpeed,
110 maxNeighbors = a.m_maxNeighbors,
111 neighborDist = a.m_neighborDist,
112 neighborElev = a.m_neighborElev,
113 timeHorizon = a.m_timeHorizon,
114 timeHorizonObst = a.m_timeHorizonObst,
115 navigationEnabled = a.m_navigationEnabled,
116 collisionEnabled = a.m_collisionEnabled,
117 layerOccupation = a.m_layerOccupation,
118 layerIgnore = a.m_layerIgnore
119 };
120 }
121 }
122 else
123 {
124 for (int i = 0; i < agentCount; i++)
125 {
126 a = m_lockedAgents[i];
127 m_maxRadius = max(m_maxRadius, a.radius);
128 pos = a.pos;
129 prefVel = a.m_prefVelocity;
130 vel = a.m_velocity;
131 m_outputAgents[i] = new AgentData()

133 index = i,
134 position = float2(pos.x, pos.z), //
135 worldPosition = pos,
136 baseline = pos.y,
137 prefVelocity = float2(prefVel.x, prefVel.z),
138 velocity = float2(vel.x, vel.z),
